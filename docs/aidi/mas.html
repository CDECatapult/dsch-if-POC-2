 95class CustomLLMClient:
 96    """Class to create custom client in Autogen."""
 97
 98    def __init__(self, config, response_model, service_settings):
 99        """CustomLLMClient init."""
100        self.response_model = (
101            response_model  # Store the model type dynamically
102        )
103        self.model = service_settings["selected_model"]
104        self.LLM_SERVICE = service_settings["service_name"]
105        if self.LLM_SERVICE == "ollama":
106            self.BASE_URL_OLLAMA_COMPATIBLE_WITH_OPENAI_API = service_settings[
107                "base_url_compatible_with_openai_api"
108            ]
109
110    def create(self, params) -> SimpleNamespace:
111        """Create CustomLLMClient."""
112        if self.LLM_SERVICE == "OpenAI":
113            client = instructor.from_openai(OpenAI())
114        elif self.LLM_SERVICE == "ollama":
115            client = instructor.from_openai(
116                OpenAI(
117                    base_url=self.BASE_URL_OLLAMA_COMPATIBLE_WITH_OPENAI_API,
118                    api_key="ollama",  # required, but unused
119                ),
120                mode=instructor.Mode.JSON,
121            )
122        else:
123            raise ValueError("Invalid LLM service name.")
124
125        try:
126            response = client.chat.completions.create(
127                model=self.model,
128                messages=params["messages"],
129                response_model=self.response_model,
130                temperature=0,
131                max_retries=5,
132            )
133        except Exception as e:
134            print("ERROR:", e)
135            raise
136
137        autogen_response = SimpleNamespace()
138        autogen_response.choices = []
139        autogen_response.model = "custom_llm_json"
140
141        choice = SimpleNamespace()
142        choice.message = SimpleNamespace()
143        # Dynamically serialize the response using the selected response model
144        choice.message.content = response.model_dump_json()
145        choice.message.function_call = None
146        autogen_response.choices.append(choice)
147        return autogen_response
148
149    def message_retrieval(self, response):
150        """Message retrieval."""
151        choices = response.choices
152        return [choice.message.content for choice in choices]
153
154    def cost(self, response) -> float:
155        """Cost."""
156        response.cost = 0
157        return 0
158
159    @staticmethod
160    def get_usage(response):
161        """Get Usage."""
162        return {}

181    def create_and_run(self, task) -> ChatResult:
182        """Creates and runs a set of agents."""
183        logger.info("Executing MultiAgentSystem create_and_run function.")
184
185        if self.full_config["group_chat"].get("structured_output"):
186            # Create planner agent.
187            planner = AssistantAgent(
188                name="Planner",
189                system_message=self.full_config["prompt"].get(
190                    "planner_system_message"
191                ),
192                llm_config={
193                    "config_list": self.custom_config_list,
194                    "cache": None,
195                    "cache_seed": None,
196                },
197            )
198
199            planner.register_model_client(
200                model_client_cls=CustomLLMClient,
201                response_model=TaskPlan,
202                service_settings=self.service_settings,
203            )
204        else:
205            # Create planner agent without custom config.
206            planner = AssistantAgent(
207                name="Planner",
208                system_message=self.full_config["prompt"].get(
209                    "planner_system_message"
210                ),
211                llm_config={
212                    "config_list": self.config_list,
213                    "cache": None,
214                    "cache_seed": None,
215                },
216            )
217
218        user_proxy = UserProxyAgent(
219            name="Admin",
220            system_message=self.full_config["prompt"].get(
221                "user_proxy_system_message"
222            ),
223            code_execution_config=False,
224        )
225
226        coder = AssistantAgent(
227            name="Coder",
228            llm_config={
229                "config_list": self.config_list,
230                "cache": None,
231                "cache_seed": None,
232            },
233            system_message=self.full_config["prompt"].get(
234                "coder_system_message"
235            ),
236        )
237        if self.full_config["group_chat"].get("structured_output"):
238            validator = AssistantAgent(
239                name="Validator",
240                llm_config={
241                    "config_list": self.custom_config_list,
242                    "cache": None,
243                    "cache_seed": None,
244                },
245                system_message=self.full_config["prompt"].get(
246                    "validator_system_message"
247                ),
248            )
249
250            validator.register_model_client(
251                model_client_cls=CustomLLMClient,
252                response_model=ValidatorOutput,
253                service_settings=self.service_settings,
254            )
255        else:
256            # Create validator agent without custom config.
257            validator = AssistantAgent(
258                name="Validator",
259                llm_config={
260                    "config_list": self.config_list,
261                    "cache": None,
262                    "cache_seed": None,
263                },
264                system_message=self.full_config["prompt"].get(
265                    "validator_system_message"
266                ),
267            )
268
269        code_executor = UserProxyAgent(
270            name="Executor",
271            system_message=self.full_config["prompt"].get(
272                "executer_system_message"
273            ),
274            human_input_mode="NEVER",  ### "ALWAYS"
275            code_execution_config={
276                "last_n_messages": 3,
277                "executor": LocalCommandLineCodeExecutor(
278                    work_dir=self.full_config["code"].get(
279                        "code_folder", "translation_folder"
280                    ),
281                    timeout=60,
282                ),
283            },
284        )
285        ## overrite validator settings if validation_by_tool is True
286        if self.full_config["group_chat"].get("validation_by_tool"):
287            validator = AssistantAgent(
288                name="Validator",
289                llm_config={
290                    "config_list": self.config_list,
291                    "cache": None,
292                    "cache_seed": None,
293                },
294                system_message=self.full_config["prompt"].get(
295                    "tool_calling_validator_system_message"
296                ),
297            )
298
299            tool_runner_agent = AssistantAgent(
300                name="function_executor_agent",
301                system_message=self.full_config["prompt"].get(
302                    "tool_runner_system_message"
303                ),
304                llm_config={
305                    "config_list": self.config_list,
306                    "cache": None,
307                    "cache_seed": None,
308                },
309            )
310
311            # Register the tool signature with the assistant agent.
312            validator.register_for_llm(
313                name=self.Name_Selected_Tool, description="Data Validator"
314            )(self.Selected_Tool)
315            # Register the tool function with the user proxy agent.
316            tool_runner_agent.register_for_execution(
317                name=self.Name_Selected_Tool
318            )(self.Selected_Tool)
319
320        def _custom_speaker_selection_func(
321            last_speaker: Agent, groupchat: GroupChat
322        ):
323            """Define a customized speaker selection function.
324
325            A recommended way is to define a transition for each speaker
326            in the groupchat.
327
328            Returns:
329                Return an `Agent` class or a string from ['auto', 'manual',
330                'random', 'round_robin'] to select a default method to use.
331            """
332            messages = groupchat.messages
333
334            if len(messages) <= 1:
335                # first, let the Coder retrieve relevant data
336                return planner
337
338            if last_speaker is planner:
339                # if the last message is from planner,
340                # let the Coder to write code
341                return coder
342
343            elif last_speaker is coder:
344                if (
345                    "```python" in messages[-1]["content"]
346                    or "python_code" in messages[-1]["content"]
347                    or "Python" in messages[-1]["content"]
348                ):
349                    return code_executor
350                else:
351                    # Otherwise, let the user_proxy to continue
352                    return user_proxy
353            elif last_speaker is code_executor:
354                if "exitcode: 1" in messages[-1]["content"]:
355                    # If the last message indicates an error,
356                    # let the coder to improve the code
357                    return coder
358                else:
359                    return validator
360
361            elif last_speaker is validator:
362                # If the last message indicates an error, go to X agent again
363                if "INVALID" in messages[-1]["content"]:
364                    return planner  ##coder #planner
365                elif messages[-1].get("tool_calls"):
366                    return tool_runner_agent
367                else:
368                    # Otherwise, go to the user
369                    return user_proxy  ## return None
370            elif self.full_config["group_chat"].get("validation_by_tool") and (
371                last_speaker is tool_runner_agent
372            ):
373                return validator  ##
374            else:
375                # default to auto speaker selection method
376                return "auto"
377
378        if self.full_config["group_chat"].get("validation_by_tool"):
379            active_agent_list = [
380                user_proxy,
381                coder,
382                validator,
383                code_executor,
384                planner,
385                tool_runner_agent,
386            ]
387        else:
388            active_agent_list = [
389                user_proxy,
390                coder,
391                validator,
392                code_executor,
393                planner,
394            ]
395
396        groupchat = GroupChat(
397            agents=active_agent_list,
398            messages=self.full_config["group_chat"].get("messages"),
399            max_round=self.full_config["group_chat"].get("max_round"),
400            speaker_selection_method=_custom_speaker_selection_func,
401            send_introductions=self.full_config["group_chat"].get(
402                "send_introductions"
403            ),
404        )
405
406        manager = GroupChatManager(
407            groupchat=groupchat,
408            llm_config={"config_list": self.config_list, "cache_seed": None},
409        )
410
411        is_cache = self.full_config["group_chat"].get("cache_chat")
412        if is_cache:
413            with Cache.disk(cache_seed=45) as cache:
414                logger.info("Initiating group chat with cache.")
415                groupchat_history_custom = user_proxy.initiate_chat(
416                    manager,
417                    message=task,
418                    cache=cache,
419                )
420        else:
421            logger.info("Initiating group chat without cache.")
422            groupchat_history_custom = user_proxy.initiate_chat(
423                manager,
424                message=task,
425            )
426        return groupchat_history_custom
